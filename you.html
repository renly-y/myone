  <script>
    const container = document.getElementById('hearts');
    const used = new Set();

    function getSafePos() {
      let gx, gy, key;
      for (let i = 0; i < 25; i++) {
        gx = Math.floor(Math.random() * 12);
        gy = Math.floor(Math.random() * 12);
        key = gx + "," + gy;
        if (!used.has(key)) {
          used.add(key);
          setTimeout(() => used.delete(key), 8000);
          return { x: gx * 8.33, y: gy * 8.33 };
        }
      }
      return { x: Math.random() * 100, y: Math.random() * 100 };
    }

    function createHeart() {
      const p = getSafePos();
      // 避开文字中心
      if (Math.abs(p.x - 50) < 34 && Math.abs(p.y - 50) < 24) return;

      const h = document.createElement('div');
      h.className = 'heart';

      const size = Math.random() * 14 + 10;
      const tx = (Math.random() - 0.5) * 140;
      const scale = (Math.random() * 0.4 + 0.8).toFixed(2);
      const dur = Math.random() * 3 + 6;

      h.style.width = h.style.height = size + 'px';
      h.style.left = p.x + '%';
      h.style.top = p.y + '%';
      h.style.setProperty('--tx', tx + 'px');
      h.style.setProperty('--scale', scale);
      h.style.animationDuration = dur + 's';
      h.style.animationDelay = Math.random() * 2 + 's';

      container.appendChild(h);
      setTimeout(() => h.remove(), dur * 1000);
    }

    setInterval(createHeart, 320);
  </script>
